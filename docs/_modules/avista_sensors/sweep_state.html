[docs]class SweepState(Enum):
    """Enum representing the various states of the sensor sweep"""

    IDLE = 0
    STARTING = 1
    INITIALIZING = 2
    EXECUTING = 3
    STOPPING = 4

    def __repr__(self):
        """An unambiguous representation of SweepState"""
        return f"Manager State: {self.name}"

    def __str__(self):
        """A readable representation of SweepState"""
        return f"{self.name}"
